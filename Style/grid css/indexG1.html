/* مثال على استخدام :root لتحديد المتغيرات */
:root {
    --primary-color: #4e54c8;
    --secondary-color: #8f94fb;
    --spacing: 1rem;
    --border-radius: 8px;
}

/* مثال على استخدام Grid */
.grid-container {
    display: grid;
    grid-template-columns: repeat(3, 1fr);
    gap: var(--spacing);
}

.grid-item {
    background-color: var(--light-color);
    border-radius: var(--border-radius);
    padding: var(--spacing);
}

.header {
    grid-column: 1 / -1; /* يمتد عبر جميع الأعمدة */
}
